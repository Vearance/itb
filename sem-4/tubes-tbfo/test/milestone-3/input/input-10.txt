program ComplexSemantic;
const
  limit == 3;

type
  ScoreArray == array[1..3] of integer;

var
  scores: ScoreArray;
  i, total: integer;
  avg: real;
  ok: boolean;

begin
  total := 0;
  i := 1;

  while i <= limit do
  begin
    scores[i] := i * 10;
    total := total + scores[i];
    i := i + 1;
  end;

  avg := total / limit;
  ok := avg >= 20.0;

  if ok then
    writeln('passed')
  else
    writeln('failed');
end.
